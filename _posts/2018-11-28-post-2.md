- Generic solution using mark for insert and print stage.

class Solution {
    public:
        vector<int> preorderTraversal(TreeNode* root) {
            vector<int> ret;
            help_func(root, ret);
            return ret;
        }

        void help_func(TreeNode* node, vector<int> &ret) {
            if (node == NULL)
                return;

            ret.push_back(node->val);
            help_func(node->left, ret);
            help_func(node->right, ret);
        }
};

class Solution {
    public:
        vector<int> preorderTraversal(TreeNode* root) {
            vector<int> ret;
            if (root == NULL)
                return ret;

            stack<TreeNode*> node_s;
            node_s.push(root);

            while (node_s.size() > 0) {
                TreeNode* node = node_s.top();
                node_s.pop();
                ret.push_back(node->val);
                if(node->right)
                    node_s.push(node->right);
                if(node->left)
                    node_s.push(node->left);
            }

            return ret;
        }
};

class Solution {
    public:
        vector<int> preorderTraversal(TreeNode* root) {
            vector<int> ret;
            if (root == NULL)
                return ret;

            stack<TreeNode*> node_s;
            TreeNode* node = root;

            while (node) {
                ret.push_back(node->val);
                if (node->right)
                    node_s.push(node->right);

                if (node->left) {
                    node = node->left;
                } else {
                    if (node_s.size()>0) {
                        node = node_s.top();
                        node_s.pop();
                    } else {
                        return ret;
                    }
                }
            }

            return ret;
        }
};

/*Generic Solution for all kinds of traversal */
class NodeStatus {
    public:
        int status; /* 0: insert, 1: print*/
        TreeNode* node;

        NodeStatus(TreeNode* n) {
            status = 0;
            node = n;
        }
};

class Solution {
    public:
        vector<int> preorderTraversal(TreeNode* root) {
            vector<int> res;
            if (root==NULL)
                return res;

            stack<NodeStatus*> list;

            NodeStatus *n_s = new NodeStatus(root);
            list.push(n_s);

            while (!list.empty()) {
                NodeStatus *n_i = list.top();
                list.pop();
                if (n_i->node == NULL)
                    continue;

                if (n_i->status == 0) {
                    list.push(new NodeStatus(n_i->node->right));
                    list.push(new NodeStatus(n_i->node->left));
                    n_i->status =1;
                    list.push(n_i);
                } else {
                    res.push_back(n_i->node->val);
                }
            }
            return res;
        }
};
