class Solution {
    public:
        vector<int> postorderTraversal(TreeNode* root) {
            vector<int> ret;
            help_func(root, ret);
            return ret;
        }

        void help_func(TreeNode* node, vector<int> &ret) {
            if (node == NULL)
                return;

            help_func(node->left, ret);
            help_func(node->right, ret);
            ret.push_back(node->val);

        }
};

/* Visiting by Root-Right-Left, then printing reversely */
class Solution {
    public:
        vector<int> postorderTraversal(TreeNode* root) {
            vector<int> ret;
            if (root==NULL)
                return ret;

            stack<TreeNode*> s1;
            stack<int> s2;

            s1.push(root);

            while (s1.size() > 0) {
                TreeNode* node = s1.top();
                s1.pop();
                if (node->left)
                    s1.push(node->left);
                if (node->right)
                    s1.push(node->right);

                s2.push(node->val);
            }

            while (s2.size()>0) {
                ret.push_back(s2.top());
                s2.pop();
            }

            return ret;
        }
};

class Solution {
    public:
        vector<int> postorderTraversal(TreeNode* root) {
            vector<int> ret;
            if (root==NULL)
                return ret;

            stack<TreeNode*> s1;

            TreeNode* node = root;


            do {
                while (node) {
                    if (node->right)
                        s1.push(node->right);

                    s1.push(node);
                    node = node->left;
                }

                /* Left-most child */
                node = s1.top();
                s1.pop();

                /* right tree hasn't been processed yet */
                if (node->right && s1.size()>0 && node->right == s1.top()) {
                    s1.pop();
                    s1.push(node);
                    node = node->right;
                } else {
                    ret.push_back(node->val);
                    node = NULL;
                }

            } while (s1.size()>0);

            return ret;
        }
};

/* Use hash_table to check left or right node on path */
class Solution {
    public:
        vector<int> postorderTraversal(TreeNode* root) {
            vector<int> res;
            if (root==NULL)
                return res;

            unordered_map<TreeNode*, int> visited;
            stack<TreeNode*> t_stack;

            TreeNode* node = root;

            while (node || !t_stack.empty()) {
                if (node != NULL) {
                    t_stack.push(node);
                    /* on left path */
                    visited[node] = 2;

                    if (node->right != NULL) {
                        /* on right path */
                        visited[node->right] = 1;
                        t_stack.push(node->right);
                    }
                    node = node->left;
                } else {
                    node = t_stack.top();
                    t_stack.pop();
                    if (visited[node] == 2) {
                        res.push_back(node->val);
                        node = NULL;
                    } else {
                        /* now searching starts on this sub-tree, this 
                           right node becomes left path node */
                        visited[node]++;
                        t_stack.push(node);

                        if (node->right) {
                            visited[node->right] = 1;
                            t_stack.push(node->right);
                        }
                        node = node->left;
                    }
                }
            }
            return res; 
        }
};

/* Generic Solution using status of node */
class NodeStatus {
    public:
        int status; /* 0: insert, 1: print*/
        TreeNode* node;

        NodeStatus(TreeNode* n) {
            status = 0;
            node = n;
        }
};

class Solution {
    public:
        vector<int> postorderTraversal(TreeNode* root) {
            vector<int> res;
            if (root==NULL)
                return res;

            stack<NodeStatus*> list;

            NodeStatus *n_s = new NodeStatus(root);
            list.push(n_s);

            while (!list.empty()) {
                NodeStatus *n_i = list.top();
                list.pop();
                if (n_i->node == NULL)
                    continue;

                if (n_i->status == 0) {
                    n_i->status =1;
                    list.push(n_i);
                    list.push(new NodeStatus(n_i->node->right));
                    list.push(new NodeStatus(n_i->node->left));
                } else {
                    res.push_back(n_i->node->val);
                }
            }
            return res;
        }
};
